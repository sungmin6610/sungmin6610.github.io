```python
from sklearn.model_selection import train_test_split

# data  나누기
data = study['공부시간']
target = study['합격여부']

# train, test 데이터 분리
x_train, x_test, y_train, y_test = train_test_split(
    data,
    target,
    test_size=0.2,
    random_state=42)
```


```python
#Step 3. 학습 후 모델을 생성하고 예측을 수행합니다

# 로지스틱회귀 모델 생성
from sklearn.linear_model import LogisticRegression

model = LogisticRegression()
# 행(row)으로 되어있는 데이터, 열(column)로 나열
훈련용_data = x_train.values.reshape(-1, 1)
테스트용_data = x_test.values.reshape(-1, 1)

# 모델 학습
model.fit(훈련용_data, y_train)


# 테스트용_data로 예측
pred = model.predict(테스트용_data)
print(pred)
```
